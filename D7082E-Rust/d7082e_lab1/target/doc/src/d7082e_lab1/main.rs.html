1//! A simple number guessing game in Rust.
2//!
3//! The program generates a random number between 1 and 100,
4//! and prompts the player to guess it. After each guess, the program
5//! provides feedback on whether the guess was too low, too high, or correct.
6
7use std::cmp::Ordering;
8use std::io;
9
10use rand::Rng;
11
12/// Prompt the user for a guess and returns it as a u32.
13/// 
14/// This function continuously reads input from standard input until a valid positive integer is provided. If invalid input is received, an error message is displayed and the user is re-prompted.
15/// 
16fn get_user_guess() -> u32 {
17    loop {
18        println!("Please input your guess (a positive number):");
19
20        let mut guess = String::new();
21
22        io::stdin()
23            .read_line(&mut guess)
24            .expect("Failed to read line");
25
26        match guess.trim().parse::<u32>() {
27            Ok(num) if num >= 1 && num <= 100 => return num,
28            _ => println!("Invalid input. Please enter a number between 1 and 100."),
29        }
30    }
31}
32
33fn main() {
34    println!("Welcome to the Number Guessing Game!");
35    println!("I have selected a number between 1 and 100.");
36    println!("Can you guess what it is?");
37
38    let secret_number = rand::thread_rng().gen_range(1..=100);
39
40    loop {
41        let guess = get_user_guess();
42
43        match guess.cmp(&secret_number) {
44            Ordering::Less => println!("Too low! Try again."),
45            Ordering::Greater => println!("Too high! Try again."),
46            Ordering::Equal => {
47                println!("Congratulations! You guessed the number {} correctly!", secret_number);
48                break;
49            }
50        }
51    }
52}
